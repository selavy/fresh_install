=== for building neovim ===
apt install libtool
apt install libtool-bin
apt install autoconf
apt install automake
apt install pkg-config
apt install unzip

$ git clone https://github.com/neovim/neovim
$ cd neovim/
$ make
$ rm -r build/
$ make clean
$ make CMAKE_BUILD_TYPE=Release
$ sudo make install
